function doPost(e) { try { const d = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); // ── 콘티 직접 내보내기 (📊 Sheets 버튼) ── if (d.type === 'conti_export') { const shName = ('콘티_' + d.contiTitle).slice(0, 30); let sh = ss.getSheetByName(shName) || ss.insertSheet(shName); sh.clearContents(); const headers = ['씬','씬제목','컷','앵글','카메라무빙','씬내용','대사/나레이션','CG','사운드','시간(초)','출연자','소품']; sh.appendRow(headers); (d.rows||[]).forEach(r => sh.appendRow([ r.씬번호, r.씬제목, r.컷순서, r.앵글, r.카메라무빙, r.씬내용, r.대사나레이션, r.CG, r.사운드, r.시간초, r.출연자, r.소품배경 ])); // 헤더 스타일 const hdr = sh.getRange(1, 1, 1, headers.length); hdr.setBackground('#00695C'); hdr.setFontColor('#FFFFFF'); hdr.setFontWeight('bold'); sh.setFrozenRows(1); } else { // ── 자동 메타데이터 동기화 ── syncSheet(ss,'프로젝트',d.projects||[],['id','name','client','stage','type','memo']); syncSheet(ss,'기획안',d.proposals||[],['id','title','projectId','createdAt']); syncSheet(ss,'콘티',d.contis||[],['id','title','projectId','genre','createdAt']); syncSheet(ss,'사전답사',d.locations||[],['id','name','address','date','manager','notes']); syncSheet(ss,'협업',d.collabs||[],['id','brand','channel','status','date','content','contact']); } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function syncSheet(ss,name,data,cols) { let sh = ss.getSheetByName(name) || ss.insertSheet(name); sh.clearContents(); sh.appendRow(cols); data.forEach(r => sh.appendRow(cols.map(c => { const v = r[c]; return typeof v==='object'?JSON.stringify(v):(v||''); }))); } // ── Gmail 협업 메일 자동 수집 (🔍 메일 자동 수집 버튼) ── function doGet(e) { const action = (e.parameter && e.parameter.action) || ''; if (action === 'gmailScan') { const query = [ 'subject:(협업 OR 광고 OR PPL OR 콜라보 OR collab OR collaboration', 'OR 유튜브 OR 인스타 OR 제안 OR 협찬 OR 스폰서)', 'newer_than:14d' ].join(' '); const threads = GmailApp.search(query, 0, 30); const results = []; threads.forEach(function(thread) { const msg = thread.getMessages().slice(-1)[0]; // 최신 메시지 const body = msg.getPlainBody().slice(0, 600); const from = msg.getFrom(); const subj = msg.getSubject(); // 채널 감지 const lower = (subj + body).toLowerCase(); let channel = 'other'; if (lower.includes('유튜브') || lower.includes('youtube')) channel = 'youtube'; else if (lower.includes('인스타') || lower.includes('instagram') || lower.includes('릴스') || lower.includes('reels')) channel = 'instagram'; else if ((lower.includes('유튜브') || lower.includes('youtube')) && (lower.includes('인스타') || lower.includes('instagram'))) channel = 'both'; results.push({ threadId: thread.getId(), subject: subj, from: from, date: msg.getDate().toISOString().slice(0,10), snippet: body, channel: channel }); }); const output = ContentService.createTextOutput( JSON.stringify({ ok: true, emails: results }) ).setMimeType(ContentService.MimeType.JSON); return output; } // 기본 응답 return ContentService.createTextOutput(JSON.stringify({ ok: true, info: 'CREYONG C.LAB Apps Script' })) .setMimeType(ContentService.MimeType.JSON); }
